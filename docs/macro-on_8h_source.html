34 // Utility function to create Sequence and Choice Expressions from an array.
35 // C and C++ treat arrays differently, so there is one initializer style for each.
36 #ifndef __PEGA_TEXTO_MACRO_ON_H__
37 #define __PEGA_TEXTO_MACRO_ON_H__
38 
39 #ifdef __cplusplus
40 # include <initializer_list>
41 # include <cstring>
42 # define __PT_EXPR_FROM_ARRAY_PARAM std::initializer_list<pt_expr *> l
43 # define __PT_EXPR_FROM_ARRAY_CAST(...) {__VA_ARGS__}
44 # define __PT_EXPR_FROM_ARRAY_GET_N \
45  N = l.size() == 1 && *l.begin() == NULL ? 0 : l.size(); \
46  pt_expr *const *nt_exprs = l.begin()
47 #else
48 # include <string.h>
49 # define __PT_EXPR_FROM_ARRAY_PARAM pt_expr **nt_exprs
50 # define __PT_EXPR_FROM_ARRAY_CAST(...) ((pt_expr*[]){__VA_ARGS__, NULL})
51 # define __PT_EXPR_FROM_ARRAY_GET_N \
52  for(aux = nt_exprs; *aux; aux++); \
53  N = aux - nt_exprs
54 #endif
55 static pt_expr *__pt_expr_from_array(pt_expr *(*f)(pt_expr **, int, uint8_t, pt_expression_action),
56  __PT_EXPR_FROM_ARRAY_PARAM,
57  uint8_t own_expressions,
58  pt_expression_action action) {
59  int N, byte_size;
60  pt_expr **aux;
61  __PT_EXPR_FROM_ARRAY_GET_N;
62  byte_size = N * sizeof(pt_expr *);
63  if(N == 0) {
64  return f(NULL, N, own_expressions, action);
65  }
66  else if(aux = (pt_expr **) malloc(byte_size)) {
67  return f((pt_expr **) memcpy(aux, nt_exprs, byte_size), N, own_expressions, action);
68  }
69  else {
70  return NULL;
71  }
72 }
73 #endif // __PEGA_TEXTO_MACRO_ON_H__
