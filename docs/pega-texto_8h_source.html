460  pt__match_action *action;
461  // Double capacity, if reached
462  if(context->action_stack.size == context->action_stack.capacity) {
463  int new_capacity = context->action_stack.capacity * 2;
464  action = (pt__match_action *) PT_REALLOC(context->action_stack.actions, new_capacity * sizeof(pt__match_action), context->opts->userdata);
465  if(action) {
466  context->action_stack.capacity = new_capacity;
467  context->action_stack.actions = action;
468  }
469  else {
470  return NULL;
471  }
472  }
473  action = context->action_stack.actions + (context->action_stack.size)++;
474  action->f = f;
475  action->str = str;
476  action->size = size;
477  action->argc = argc;

482 static void pt__run_actions(pt__match_context *context, pt_match_result *result) {
483  PT_DATA *data_stack;
484  if(sizeof(PT_DATA) > sizeof(pt__match_action)) {
485  // Allocate the data stack
486  data_stack = (PT_DATA *) PT_MALLOC(context->action_stack.size * sizeof(PT_DATA), context->opts->userdata);
487  if(data_stack == NULL) {
488  result->matched = PT_NO_STACK_MEM;
489  return;
490  }
491  }
492  else {
493  // PT_DATA may safely overwrite popped actions memory, as the later will be read just before pushing results
494  // This avoids a malloc/free pair
495  data_stack = (PT_DATA *) context->action_stack.actions;
496  }
497 
498  // index to current Data on the stack
499  int data_index = 0;
500 
501  // Fold It, 'til there are no Actions left.
502  // Note that this only works because of how the Actions are layed out in
503  // the Action Stack.
504  pt__match_action *action;
505  for(action = context->action_stack.actions; action < context->action_stack.actions + context->action_stack.size; action++) {
506  // "pop" arguments
507  data_index -= action->argc;
508  // run action with arguments (which are still stacked in `data_stack` in the right position)
509  data_stack[data_index] = action->f(
510  action->str,
511  action->size,
512  action->argc,
513  data_stack + data_index,
514  context->opts->userdata
515  );
516  // "push" result
517  data_index++;
518  }
519  result->data = data_stack[0];
520  if(sizeof(PT_DATA) > sizeof(pt__match_action)) {
521  PT_FREE(data_stack, context->opts->userdata);
522  }
523 }

561 static pt__match_expr_result pt__match_expr(pt__match_context *context, const pt_expr *const e, pt_element_string sp) {
562  pt__match_expr_result result = { 0, 0, 1 };
563  switch(e->op) {
564  case PT_OP_END: {
565  result.success = 1;
566  break;
567  }
568 
569  case PT_OP_ELEMENT: {
570  result.success = (*sp) == (PT_ELEMENT_TYPE) e->element;
571  result.sp_advance = 1;
572  break;
573  }
574 
575  case PT_OP_LITERAL: {
576  result.success = strncmp(sp, e->str, e->N) == 0;
577  result.sp_advance = e->N;
578  break;
579  }
580 
581  case PT_OP_CASE_INSENSITIVE: {
582  result.success = strncasecmp(sp, e->str, e->N) == 0;
583  result.sp_advance = e->N;
584  break;
585  }
586 
587  case PT_OP_CHARACTER_CLASS: {
588  result.success = e->character_class_matcher(*sp) != 0;
589  result.sp_advance = 1;
590  break;
591  }
592 
593  case PT_OP_SET: {
594  result.success = *sp && strchr(e->str, *sp);
595  result.sp_advance = 1;
596  break;
597  }
598 
599  case PT_OP_RANGE: {
600  PT_ELEMENT_TYPE element = *sp;
601  result.success = element >= PT_RANGE_UNPACK_FROM(e->range) && element <= PT_RANGE_UNPACK_TO(e->range);
602  result.sp_advance = 1;
603  break;
604  }
605 
606  case PT_OP_ANY: {
607  result.success = (*sp) != 0;
608  result.sp_advance = 1;
609  break;
610  }
611 
612  case PT_OP_CUSTOM_MATCHER: {
613  int custom_matcher_result = e->matcher(sp, context->opts->userdata);
614  result.success = custom_matcher_result > 0;
615  result.sp_advance = custom_matcher_result;
616  break;
617  }
618 
619  case PT_OP_NON_TERMINAL:
620  return pt__match_rule(context, e->index, sp);
621 
622  case PT_OP_AT_LEAST: {
623  unsigned int counter = 0;
624  pt__match_expr_result subresult;
625  while(1) {
626  subresult = pt__match_sequence(context, e, sp + result.sp_advance);
627  if(subresult.success > 0 && subresult.sp_advance > 0) {
628  result.sp_advance += subresult.sp_advance;
629  counter++;
630  }
631  else {
632  break;
633  }
634  }
635  if(subresult.success >= 0) {
636  result.success = counter >= e->quantifier;
637  result.e_advance = 1 + e->N;
638  }
639  else {
640  result.success = subresult.success;
641  }
642  break;
643  }
644 
645  case PT_OP_AT_MOST: {
646  pt__match_expr_result subresult;
647  for(unsigned int counter = 0; counter < e->quantifier; counter++) {
648  subresult = pt__match_sequence(context, e, sp + result.sp_advance);
649  if(subresult.success > 0 && subresult.sp_advance > 0) {
650  result.sp_advance += subresult.sp_advance;
651  }
652  else {
653  break;
654  }
655  }
656  if(subresult.success >= 0) {
657  result.success = 1;
658  result.e_advance = 1 + e->N;
659  }
660  else {
661  result.success = subresult.success;
662  }
663  break;
664  }
665 
666  case PT_OP_NOT: {
667  size_t current_action_count = context->action_stack.size;
668  result = pt__match_sequence(context, e, sp);
669  if(result.success >= 0) {
670  context->action_stack.size = current_action_count; // ignore captures
671  result.success = !result.success;
672  result.sp_advance = 0;
673  }
674  break;
675  }
676 
677  case PT_OP_AND: {
678  size_t current_action_count = context->action_stack.size;
679  result = pt__match_sequence(context, e, sp);
680  context->action_stack.size = current_action_count; // ignore captures
681  result.sp_advance = 0;
682  break;
683  }
684 
685  case PT_OP_SEQUENCE:
686  return pt__match_sequence(context, e, sp);
687 
688  case PT_OP_CHOICE: {
689  for(int i = 0; result.success == 0 && i < e->N; i += result.e_advance) {
690  result = pt__match_expr(context, e + 1 + i, sp);
691  }
692  result.e_advance = 1 + e->N;
693  break;
694  }
695 
696  case PT_OP_ACTION: {
697  size_t previous_action_count = context->action_stack.size;
698  result = pt__match_sequence(context, e, sp);
699  if(result.success > 0) {
