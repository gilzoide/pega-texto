32 double getCPUTime( )
33 {
34 #if defined(_WIN32)
35  /* Windows -------------------------------------------------- */
36  FILETIME createTime;
37  FILETIME exitTime;
38  FILETIME kernelTime;
39  FILETIME userTime;
40  if ( GetProcessTimes( GetCurrentProcess( ),
41  &createTime, &exitTime, &kernelTime, &userTime ) != -1 )
42  {
43  SYSTEMTIME userSystemTime;
44  if ( FileTimeToSystemTime( &userTime, &userSystemTime ) != -1 )
45  return (double)userSystemTime.wHour * 3600.0 +
46  (double)userSystemTime.wMinute * 60.0 +
47  (double)userSystemTime.wSecond +
48  (double)userSystemTime.wMilliseconds / 1000.0;
49  }

61  if ( clock_getcpuclockid( 0, &id ) == -1 )
62 #endif

64  /* Use known clock id for AIX, Linux, or Solaris. */
