66  int is_empty = argc <= 0;
67  Cons *list = create_cons_cell(is_empty ? NULL : argv[0].p, NULL, is_empty);

77 int main(int argc, char **argv) {
78  assert(argc > 1 && "Need a string to parse!");
79 
80  /* LispCode <- Spaces Sexp* !.
81  * Sexp <- (List / Atom) Spaces
82  * List <- "(" Spaces Sexp* ")"
83  * Atom <- (!("(" / ")" / Space) .)+
84  * Spaces <- Space*
85  * Space <- Comment / \s
86  * Comment <- ";" [^\n]*
87  */
88  pt_rule R[] = {
89  { "LispCode", SEQ(V("Spaces"), Q(V("Sexp"), 0), NOT(ANY)) },
90  { "Sexp", SEQ(OR(V("List"), V("Atom")), V("Spaces")) },
91  { "List", SEQ_(list, L("("), V("Spaces"), Q(V("Sexp"), 0), L(")")) },
92  { "Atom", Q_(atom, BUT(OR(S("()"), V("Space"))), 1) },
93  { "Spaces", Q(V("Space"), 0) },
94  { "Space", OR(V("Comment"), F(isspace)) },
95  { "Comment", SEQ(L(";"), Q(BUT(L("\n")), 0)) },
96  { NULL, NULL },
97  };
98  pt_grammar *g = pt_create_grammar(R, 0);
99  pt_validate_grammar(g, PT_VALIDATE_ABORT);
100 
101  pt_match_result res = pt_match_grammar(g, argv[1], NULL);
102  if(res.matched >= 0) {
103  Cons *list = res.data.p;
104  print_cons_list(list);
105  printf("\n");
106  destroy_cons_list(list);
107  }
108  else {
109  puts("Invalid string");
110  }
111 
112  pt_destroy_grammar(g);
113  return 0;
114 }
