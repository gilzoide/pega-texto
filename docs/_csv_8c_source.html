23  pt_rule R[] = {
24  { "CSV", Q(V("Line"), 0) },
25  { "Line", SEQ(V("Field"), Q(SEQ(L(","), V("Field")), 0), OR(V("EOL"), NOT(ANY))) },
26  { "Field", OR(Q(BUT(S("\",\r\n")), 1), V("Quoted")) },
27  { "Quoted", SEQ(L("\""), Q(OR(L("\"\""), BUT(L("\""))), 0), L("\"")) },
28  { "EOL", SEQ(Q(L("\r"), -1), L("\n")) },
29  { NULL, NULL },
30  };
31  pt_grammar *g = pt_create_grammar(R, 0);

33  assert(pt_match_grammar(g, str, NULL).matched >= 0 && "Error on parse: is the CSV file valid?");
