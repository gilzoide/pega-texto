24 int main(int argc, char **argv) {
25  assert(argc > 1 && "Need a file to search for includes");
26  char *str = readfile(argv[1]);
27 
28  /* Axiom <- (Include / .)*
29  * Include <- "#" Space "include" Space (Angled / Quoted)
30  * Angled <- "<" [^>]+ ">"
31  * Quoted <- "\"" [^"]+ "\""
32  * Space <- \s*
33  */
34  pt_rule R[] = {
35  { "Axiom", Q(OR(V_(print_includes, "Include"), ANY), 0) },
36  { "Include", SEQ(L("#"), V("Space"), L("include"), V("Space"), OR(V("Angled"), V("Quoted"))) },
37  { "Angled", SEQ(L("<"), Q(BUT(L(">")), 1), L(">")) },
38  { "Quoted", SEQ(L("\""), Q(BUT(L("\"")), 1), L("\"")) },
39  { "Space", Q(F(isspace), 0) },
40  { NULL, NULL },
41  };
42  pt_grammar *g = pt_create_grammar(R, 0);
43 
44  int include_count = 0;
45  pt_match_options opts = { .userdata = &include_count };
46  assert(pt_match_grammar(g, str, &opts).matched >= 0 && "Error on parse: is the string valid?");
47  printf("Found %d includes\n", include_count);
48 
49  pt_destroy_grammar(g);
50  free(str);
51  return 0;
52 }
