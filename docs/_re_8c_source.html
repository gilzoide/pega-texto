56  int i;

224 pt_grammar *create_grammar_from_string(const char *str, ...) {
225  /* Grammar <- S Definition+ !.
226  * Definition <- Identifier Arrow Exp
227  *
228  * Exp <- Seq ("/" S Seq)*
229  * Seq <- Prefix+
230  * Prefix <- ([&!] S)? Suffixed
231  * Suffixed <- Primary S Suffix?
232  * Suffix <- [+*?] S
233  * / "^" Number
234  *
235  * Primary <- "(" S Exp ")" S
236  * / "{" S Exp "}" S
237  * / Defined
238  * / Literal
239  * / Class
240  * / "." S
241  * / Identifier !Arrow
242  *
243  * Literal <- '"' (!'"' Character)* '"' S
244  * / "'" (!"'" Character)* "'" S
245  * Class <- "[" "^"? Item (!"]" Item)* "]" S
246  * Item <- Defined / Range / .
247  * Range <- . "-" [^]]
248  * Character <- "\\" [abfnrtv'"\[\]\\]
249  * / "\\" [0-2][0-7][0-7]
250  * / "\\" [0-7][0-7]?
251  * / .
252  * Defined <- "\\" [wWaAcCdDgGlLpPsSuUxX] S
253  *
254  * S <- (\s / "#" [^\n]*)*
255  * Identifier <- [A-Za-z_][A-Za-z0-9_-]* S
256  * Arrow <- "<-" S
257  * Number <- [+-]? \d+ S
258  */
259  pt_rule rules[] = {
260  { "Grammar", SEQ_(build_grammar, V("S"), Q(V("Definition"), 1), NOT(ANY)) },
261  { "Definition", SEQ(V("Identifier"), V("Arrow"), V("Exp")) },
262 
263  { "Exp", SEQ_(maybe_choice, V("Seq"), Q(SEQ(L("/"), V("S"), V("Seq")), 0)) },
264  { "Seq", Q_(maybe_seq, V("Prefix"), 1) },
265  { "Prefix", SEQ_(maybe_prefix, Q(SEQ(S("&!"), V("S")), -1), V("Suffixed")) },
266  { "Suffixed", SEQ_(maybe_quantifier, V("Primary"), Q(V("Suffix"), -1)) },
267  { "Suffix", OR(SEQ(S_(quantifier, "+*?"), V("S")), SEQ(L("^"), V("Number"))) },
268 
269  { "Primary", OR(SEQ(L("("), V("S"), V("Exp"), L(")"), V("S")),
270  SEQ_(with_action, L("{"), V("S"), V("Exp"), L("}"), V("S")),
271  V("Defined"),
272  V("Literal"),
273  V("Class"),
274  SEQ_(create_any, L("."), V("S")),
275  SEQ(V_(non_terminal, "Identifier"), NOT(V("Arrow")))) },
276  { "Literal", OR_(literal,
277  SEQ(L("\""), Q(SEQ(NOT(L("\"")), V("Character")), 0), L("\""), V("S")),
278  SEQ(L("\'"), Q(SEQ(NOT(L("\'")), V("Character")), 0), L("\'"), V("S"))) },
279  { "Class", SEQ_(char_class, L("["), Q(L("^"), -1), V("Item"), Q(SEQ(NOT(L("]")), V("Item")), 0), L("]"), V("S")) },
280  { "Item", OR(V("Defined"), V("Range"), V_(one_char, "Character")) },
281  { "Range", SEQ_(range, ANY, L("-"), BUT(L("]"))) },
282  { "Character", OR_(character,
283  SEQ(L("\\"), S("abfnrtv\'\"[]\\")),
284  SEQ(L("\\"), R("02"), R_(forward_character, "07"), R_(forward_character, "07")),
285  SEQ(L("\\"), R("07"), Q(R_(forward_character, "07"), -1)),
286  ANY) },
287  { "Defined", SEQ(L("\\"), S_(defined, "wWaAcCdDgGlLpPsSuUxX"), V("S")) },
288 
289  { "S", Q(OR(F(isspace), SEQ(L("#"), Q(BUT(L("\n")), 0))), 0) },
290  { "Identifier", SEQ(SEQ_(identifier, OR(F(isalpha), L("_")), Q(OR(F(isalnum), S("_-")), 0)), V("S")) },
291  { "Arrow", SEQ(L("<-"), V("S")) },
292  { "Number", SEQ_(tonumber, Q(S("+-"), -1), Q(F(isdigit), 1), V("S")) },
293  { NULL, NULL },
294  };
295  pt_grammar *re = pt_create_grammar(rules, 0);
296  pt_validate_grammar(re, PT_VALIDATE_ABORT);
297 
298  // Expression Actions
299  va_list actions;
300  va_start(actions, str);
301  pt_match_options opts = { .userdata = &actions };
302 
303  pt_match_result res = pt_match_grammar(re, str, &opts);
304  pt_grammar *g = NULL;
305  if(res.matched >= 0) {
306  g = res.data.p;
307  }
308 
309  va_end(actions);
310  pt_destroy_grammar(re);
311  return g;
312 }
